```go
package httpclient

import (
    "fmt"
    "io"
    "net/http"
    "time"
)

type Client struct {
    httpClient *http.Client
    userAgent  string
}

func New() *Client {
    return &Client{
        httpClient: &http.Client{Timeout: 3 * time.Second},
        userAgent:  "ccw/0.1",
    }
}

func (c *Client) Get(url string) ([]byte, error) {
    req, err := http.NewRequest("GET", url, nil)
    if err != nil {
        return nil, err
    }
    req.Header.Set("User-Agent", c.userAgent)

    resp, err := c.httpClient.Do(req)
    if err != nil {
        return nil, err
    }
    defer resp.Body.Close()

    if resp.StatusCode != http.StatusOK {
        return nil, fmt.Errorf("HTTP %d", resp.StatusCode)
    }

    return io.ReadAll(resp.Body)
}
```

Use the synodic month algorithm: days since known new moon (Jan 6, 2000) mod 29.53059. Map to 8 phases. Each phase has an emoji and name.

```go
package widgets

import (
    "fmt"
    "math"
    "time"

    "github.com/warunacds/ccstatuswidgets/internal/protocol"
)

type MoonWidget struct{}

func (w *MoonWidget) Name() string { return "moon" }

var moonPhases = []struct {
    emoji string
    name  string
}{
    {"🌑", "new moon"},
    {"🌒", "waxing crescent"},
    {"🌓", "first quarter"},
    {"🌔", "waxing gibbous"},
    {"🌕", "full moon"},
    {"🌖", "waning gibbous"},
    {"🌗", "last quarter"},
    {"🌘", "waning crescent"},
}

const synodicMonth = 29.53059

func moonPhaseForDate(t time.Time) int {
    // Known new moon: January 6, 2000
    ref := time.Date(2000, 1, 6, 0, 0, 0, 0, time.UTC)
    days := t.Sub(ref).Hours() / 24
    cycle := math.Mod(days, synodicMonth)
    if cycle < 0 {
        cycle += synodicMonth
    }
    return int(cycle / (synodicMonth / 8))
}

func (w *MoonWidget) Render(input *protocol.StatusLineInput, cfg map[string]interface{}) (*protocol.WidgetOutput, error) {
    phase := moonPhaseForDate(time.Now())
    p := moonPhases[phase%8]
    return &protocol.WidgetOutput{
        Text:  fmt.Sprintf("%s %s", p.emoji, p.name),
        Color: "dim",
    }, nil
}
```

```python
# python-sdk/ccstatuswidgets/sdk.py
import json
import sys

def widget(name):
    def decorator(func):
        def wrapper():
            try:
                input_data = json.loads(sys.stdin.read())
                config = input_data.get("_widget_config", {})
                result = func(input_data, config)
                json.dump(result, sys.stdout)
            except Exception as e:
                print(f"Plugin error: {e}", file=sys.stderr)
                sys.exit(1)
        wrapper()
        return func
    return decorator
```
